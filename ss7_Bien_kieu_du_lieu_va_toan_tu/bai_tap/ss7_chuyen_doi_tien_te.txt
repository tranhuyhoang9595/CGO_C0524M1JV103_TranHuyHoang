<script>
    // đặt tên hàm (function) và tên biến => camelCase
    function tinhToan() {
        // khai báo biến : var ,let , const
        let money = document.getElementById("amount").value;
        let from = document.getElementById("from").value;
        let to = document.getElementById("to").value;
        let result = 0;
        if (from === "USD" && to === "VND") {
            result = money * 24000;
        } else if (from === "VND" && to === "USD") {
            result = money / 24000;
        } else {
            result = money;
        }
        document.getElementById("result").innerText = "Kết quả là : " + result

    }

    // format code => Ctrl + Alt + L
</script>
